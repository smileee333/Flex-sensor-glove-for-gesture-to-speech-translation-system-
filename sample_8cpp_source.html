62template<typename max_sensor_cnt_data_type>
63void sample_space<max_sensor_cnt_data_type>::takeSample(){
64 //following loop shall update the selector pin values and read the selected flex-sensor output and store it for each flex sensor.
65 for(max_sensor_cnt_data_type i= 0;i<sensor_cnt;i++){
66 // Selecting the appropriate selector lines for capturing all the flex sensors input one by one
67 // since in Arduino design, (D5-D2) Digital GPIOs are selected for MSB-LSB of selector lines, and PORTD constitutes pins from D7-D0, So we need to left shift the i by 2 positions, so that we only account for D5-D2.
68 SEL_PORT=SEL_PORT | (i<<2);
69 // Taking the sampled data from selected flex sensor and storing in current_sample_data;
70 current_sample_data.sensor_bit[i]=digitalRead(flex_sensor_output_pin);
71 }
72}
