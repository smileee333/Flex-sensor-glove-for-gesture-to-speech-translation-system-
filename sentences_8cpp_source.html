24/**
25 * @brief Method to add a new word into existing sentence queue data-structure.
26 * @tparam max_sensor_cnt_data_type : It represents the datatype of gesture id and space id representing a word.
27 * @param input_word : It is the word to be added into sentence queue data-structure.
28 * @see sample_ids for structure description.

30template<typename max_sensor_cnt_data_type>
31void sentence_queue<max_sensor_cnt_data_type>::addWord(sample_ids<max_sensor_cnt_data_type> input_word){
32
33 //creating a DL_node dynamically for new word to be added into sentence queue data structure
34 DL_node<sentence_data<max_sensor_cnt_data_type>>* temp_word_node = new DL_node<sentence_data<max_sensor_cnt_data_type>>;
35
36 //assigning the data to dynamically created new word
37 temp_word_node->data.id.space_id=input_word.space_id;
38 temp_word_node->data.id.gesture_id=input_word.gesture_id;
39 temp_word_node->data.id_string=gesture_space_bank.extractGestureString(2,input_word.gesture_id);//since we want to add the string corresponding to speech gestures here, so we kept arr_id=2
40 temp_word_node->next=NULL;//as this node is added in the last of sentence queue data-structure so next address of last node should be NULL
41
42 // adding the dynamically created new word to sentence queue
43 // Case I: when sentence queue is not empty initially
44 if(!(head_ptr==NULL)){
45 //linkng this node to the last sentence queue end.
46 temp_word_node->previous=tail_ptr;
47 //linking sentence queue end to dynamically created node.
48 tail_ptr->next=temp_word_node;
49 //changing the sentence end marking pointer to newly added node.
50 tail_ptr=tail_ptr->next;
51 }
52 //case II: when sentence queue is empty initially
53 else{
54 //First word node in the sentence queue will not have any previous node to be pointed to.
55 temp_word_node->previous=NULL;
56 head_ptr=temp_word_node;
57 //initially both head_ptr and tail_ptr should point out to same word node as there is only one word node present.
58 tail_ptr=temp_word_node;
59 }
60 //subsequently also move the ctrl ptr to the last newly added word node.
61 ctrl_ptr=tail_ptr;
62 //since new word has been added to assigning the temporarily created node
63 temp_word_node = NULL
64 //increasing the word count
65 word_cnt++;
66 //Displaying the newly added word in display
67 //Since tail_ptr points to last newly added node, so we are reading the string from there.
68 display.printInDisplay(tail_ptr->data.id_string);
69}

136/**
137 * @brief Method to delete a word in the existing sentence queue data-structure such that word to be deleted is pointed by ctrl pointer and correspondingly the word string is updated on display.
138 * @tparam max_sensor_cnt_data_type : It is used to represent the data-type of gesture id and space id representing a word in the sentence.

142template<typename max_sensor_cnt_data_type>
143void sentence_queue<max_sensor_cnt_data_type>::deleteWord(sample_ids<max_sensor_cnt_data_type> input_word){
144 //case I: no word in the queue
145 if(ctrl_ptr==NULL){
146 //Displaying the message in display *NO WORD TO DEL*
147 display.printInDisplay("NO WORD TO DEL");
148 }
149 //case II: single word in the queue
150 elseif(head_ptr==tail_ptr){
151 //assigning tail_ptr to NULL, first so that it doesn't become dangling after release of heap memory
152 tail_ptr=NULL;
153 ctrl_ptr=tail_ptr;
154 //release the heap memory
155 delete head_ptr;
156 //assigning head_ptr to NULL, so that head_ptr doesn't become dangling
157 head_ptr=NULL;
158 //decreasing the word count
159 word_cnt--;
160 }
161 //case III: multiple words in the queue
162 else{
163 //case I: when word is in the middle of sentence:
164 if((ctrl_ptr->next!=NULL) && (ctrl_ptr->previous!=NULL)){
165 //a temporary pointer creation for deletion of word node pointed by the ctrl_ptr
166 DL_node<sentence_data<max_sensor_cnt_data_type>>* temp_word_ptr = ctrl_ptr->previous;
167 temp_word_ptr->next=ctrl_ptr->next;
168 temp_word_ptr=ctrl_ptr->next;
169 temp_word_ptr->previous=ctrl_ptr->previous;
170 display.printInDisplay(ctrl_ptr->data.id_string);
171
172 }
173 //case II: when word is in the end of sentence:
174 elseif(ctrl_ptr->next==NULL){
175 DL_node<sentence_data<max_sensor_cnt_data_type>>* temp_word_ptr=ctrl_ptr->previous;
176 temp_word_ptr->next=NULL;
177 ctrl_ptr->previous=NULL;
178 display.printInDisplay(ctrl_ptr->data.id_string)
179 }
180 //case III: when word is in the start of sentence:
181 else{
182 DL_node<sentence_data<max_sensor_cnt_data_type>>* temp_word_ptr=ctrl_ptr->next;
183 temp_word_ptr->previous=NULL;
184 ctrl_ptr->next-NULL;
185 display.printInDisplay(ctrl_ptr->data.id_string)
186 }
187 delete ctrl_ptr;
188 ctrl_ptr=temp_word_ptr;
189 word_cnt--;decreasing the word count
190 }
191}

203void sentence_queue<max_sensor_cnt_data_type>::deleteSentence(sample_ids<max_sensor_cnt_data_type> input_word){
204 //case I: when there is no word in sentence
205 if(head_ptr==NULL){
206 //Display message in display *NO SENTENCE*

235 * @brief Method to play the audio file for each word in the acquired sentence one by one such that whole acquired sentence is translated finally.
236 * @tparam max_sensor_cnt_data_type : It is used to represent the data-type of gesture id and space id representing a word in the sentence.

240template<typename max_sensor_cnt_data_type>
241void sentence_queue<max_sensor_cnt_data_type>::translateGesture(sample_ids<max_sensor_cnt_data_type> input_word){
242 //stopping of Timer - Timer Destructor call
243 ~samplingTimer;
244
245 //translating each gesture word one by one to speech
246 DL_node<sentence_data<max_sensor_cnt_data_type>>* tmp_ptr=NULL;
247 sentence_data<max_sensor_cnt_data_type> tmp_data;
248
249 while(word_cnt!=0){
250 //a. extraction of gesture_id and space_id as well as gesture_string from the word node pointed by head_ptr.
251 tmp_data= {{head_ptr->data.id.space_id,head_ptr->data.id.gesture_id},head_ptr->data.id_string};
252 //b. removing the word node pointed by head_ptr from sentence queue and move the head_ptr to next word in the sentence.
253 tmp_ptr=head_ptr;
254 head_ptr=head_ptr->next;
255 delete tmp_ptr;
256 tmp_ptr=NULL;
257 //if the current word to be translated is last word then also assign the tail node to next. In this way both tail and head_ptr will be pointing to NULL. consequently ctrl_ptr should also point to
258 //tail_ptr.
259 if(word_cnt==1){
260 tail_ptr=head_ptr;
261 ctrl_ptr=tail_ptr;
262 }
263 //invoking the translator method to wait until Translator is available.
264 translator.waitUntilTranslatorAvailable();
265 //Displaying the string corresponding of tmp_data (tmp_data.id_string)
266 display.printInDisplay(tmp_data.id_string);
267
268 //invoking the translator method to translate the word to speech.
269 translator.translateGesture(tmp_data.id);
270 word_cnt--;
271 }
272}
