19/**
20 * @brief Constructor of Timer class: It is used to initialize and start the timer with the given configuration parameters
21 * @tparam preScaleVal : prescale value for timer counting
22 * @tparam compare_reg : compare register to be used for compare match interrupt
23 * @tparam compare_val : value to be loaded in the compare register
24*/
25template< prescale_value preScaleVal, compare_reg_enable compare_reg, uint16_t compare_val>
26Timer::Timer(){
27 //Following steps taken for timer initialization and start------
28 //a. stopping the global interrupts from interrupting the initialization process
29 cli();
30 //b. To stop the timer from counting -> if counting previously and making everything else disabled
31 TCCRxA_ref = 0;
32 TCCRxB_ref = 0;
33 //c. To enable the compare match interruption mode for the timer
34 TIMSKx_ref |= TIMSKx_value;
35 //d. To define the compare value for the timer
36 OCRxX_ref = OCRxX_value;
37 //e. To reset the Timer count value to 0 as well shall start counting from Zero here
38 TCNTx_ref = 0;
39 //f. To update the sampling status
40 sampleData.setSamplingStatus(True);//True means now sampling has been started
41 //g. To define the prescaler value. Defining prescaler value will also start the timer counting
42 TCCRxB_ref |= TCCRxB_value;
43 //h. To re-enable the global interrupts as Timer has been started and initialized now
44 sei();
45}
