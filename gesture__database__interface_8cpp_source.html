28 // a. Gesture list extraction and assignment case for gesture space with input gesture space id

42 // b. Gesture space entry list preparation and assignment case
43 elseif (gesture_input.space_id==meta_info_interface_val.space_cnt-1){
44 // To create a dynamic array of size meta_info_interface_val.space_cnt-1
45 gesture* gesture_array_ptr=new gesture[meta_info_interface_val.space_cnt-1];

121template<typename max_sensor_cnt_data_type>
122 void max_sensor_cnt_data_type::getSpeechSpaceGestureCntArray(max_sensor_cnt_data_type* arr_ptr){
123 // a. creating an array dynammically and assign to input arr_ptr with size of total number of speech gesture spaces
124 // substraction of last space_id and 1st speech space id gives total number of speech spaces available.
125 arr_ptr= new max_sensor_cnt_data_type[(meta_info_interface_val.space_cnt-1)-(speech_start_idx)];
126 // b. assigning the total gesture count to each element of above created array
127 //since, array needs to start from 0 idx, as we are only referring to speech spaces, so keep i-speech_start_idx
128 for(max_sensor_cnt_data_type i=speech_start_idx;i<(meta_info_interface_val.space_cnt-1);i++){
129 arr_ptr[i-speech_start_idx]=pgm_read_word(&database_interface_ptr->data_database[i].cnt);
130 }
131}
