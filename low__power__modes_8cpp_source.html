32 * @brief To configure the CPU as per input power mode and accordingly set the current power mode

34 * @see power_mode for enum data-type description

36void powerMode::setPowerMode(power_mode input_mode){
37 //case I: when input power mode is deep sleep mode
38 if(input_mode==deep_sleep){
39 //Configuring CPU to go to deep sleep mode
40 static byte prevADCSRA= ADCSRA;
41 // a. disabling the ADC
42 ADCSRA = 0;
43 // b. setting up sleep mode to SLEEP_MODE_PWR_DOWN to enable maximum power down.
44 set_sleep_mode(SLEEP_MODE_PWR_DOWN);
45 // c. Enabling the sleep
46 sleep_enable;
47 // e. To Turn off burn out detection
48 MCUCR = bit (BODS) | bit (BODSE);
49 MCUCR = bit (BODS);
50 // Setting current mode to be deep_sleep mode
51 current_mode=deep_sleep;
52 }
53 else if(input_mode==idle_sleep){
54 //case II: when input power mode is idle sleep mode
55 //Configuring CPU to go to idle sleep mode
56 // a. setting up the sleep mode to SLEEP_MODE_IDLE
57 set_sleep_mode(SLEEP_MODE_IDLE);
58 // b. enabling the sleep
59 sleep_enable;
60 // Setting current mode to be sleep idle mode
61 current_mode=idle_sleep;
62 }
63 else if(input_mode==no_sleep){
64 // Disabling the sleep mode until next time we go to it deliberately through code
65 sleep_disable();
66 }
67}
